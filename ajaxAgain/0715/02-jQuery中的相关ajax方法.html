$.ajax可以发get请求,可以发post请求,还可以发jsonp请求 $.ajax({ type:请求方式 get或者post,默认get, url:请求地址, data:请求数据,取值 对象 或者 参数字符串, contentType: 设置传递参数的格式 默认取值 application/x-www-form-urlencoded 取值也可以写 application/json, dataType:"jsonp", jsonp:"callback参数名", jsonpCallback:"callback对应的函数名", success:function( res ){ console.log( res ); }, error:function( xhr ){ console.log( xhr ); }, beforeSend:function(){ return false } }); $.get( 请求地址 [,请求参数] , 成功的回调函数 ); 发起get请求 $.post( 请求地址 [,请求参数] , 成功的回调函数 ); 发起post请求 $.getJSON( json文件所在路径, 成功的回调函数 );
